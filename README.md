## Install

```bash
npm install --save hi-viz
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'hi-viz'
import 'hi-viz/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
